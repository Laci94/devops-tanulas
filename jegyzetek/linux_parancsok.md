# Linux parancsok

## Navigáció
pwd     = megmutatja hol vagyok
ls      = fájlok listája
ls -la  = részletes lista
cd ~    = home mappába ugrás
cd ..   = egy mappával feljebb

## Fájlok
touch fajl.txt    = üres fájl létrehozása
mkdir mappa       = mappa létrehozása
cp fajl cél       = másolás
mv fajl cél       = mozgatás vagy átnevezés
rm fajl           = törlés
cat fajl          = fájl tartalmának kiírása
